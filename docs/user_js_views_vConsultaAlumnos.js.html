        <div><strong>Nombre del Padre:</strong> ${alumno.nombrePadre || 'Por determinar'} ${alumno.apellidosPadre || ''}</div>
        <div><strong>TelÃ©fono:</strong> ${alumno.telefono}</div>
    `;
    modal.show();
}
